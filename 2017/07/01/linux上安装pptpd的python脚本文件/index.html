一个可以在linux上安装pptpd的python脚本文件

必备技能

python脚本文件，linux的centos和ubuntu 一键安装pptpd vpn服务器及其客户端

代码如下：

一.centos下的pptpd服务器代码：

  • 1.代码如下：

import subprocess
subprocess.call(‘yum install ppp -y’,shell=True)
subprocess.call(‘rpm -Uvh http://poptop.sourceforge.net/yum/stable/rhel6/pptp-release-current.noarch.rpm',shell=True)
subprocess.call(‘yum install pptpd -y’,shell=True)

with open(“/etc/pptpd.conf”, “r”) as f1:
d1 = f1.readlines()
with open(“/etc/pptpd.conf”, “w”) as f_w1:
for line in d1:
line = line.replace(“#localip 192.168.0.1”, “localip 192.168.1.1”)
line = line.replace(“#remoteip 192.168.0.234-238,192.168.0.245”, “remoteip 192.168.1.2-238”)
f_w1.write(line)

with open(“/etc/ppp/options.pptpd”, “r”) as f2:
d2 = f2.readlines()
with open(“/etc/ppp/options.pptpd”, “w”) as f_w2:
for line in d2:
line = line.replace(“#ms-dns 10.0.0.1”, “ms-dns 8.8.8.8”)
line = line.replace(“#ms-dns 10.0.0.2”, “ms-dns 4.4.4.4”)
f_w2.write(line)

with open(“/etc/sysctl.conf”, “r”) as f4:
d4 = f4.readlines()
with open(“/etc/sysctl.conf”, “w”) as f_w4:
for line in d4:
line = line.replace(“net.ipv4.ip_forward = 0”, “net.ipv4.ip_forward = 1”)
f_w4.write(line)

subprocess.call(‘sysctl -p’,shell=True)

f3=open(“/etc/ppp/chap-secrets”,’a’)
for i in range(1,101):
f3.write(“user”+str(i)+” pptpd”+” user”+str(i)+”yourmima”+” *”+”\n”)

subprocess.call(‘iptables -D INPUT -j REJECT –reject-with icmp-host-prohibited’,shell=True)
subprocess.call(‘iptables -D FORWARD -j REJECT –reject-with icmp-host-prohibited’,shell=True)
subprocess.call(‘iptables -A INPUT -i lo -p all -j ACCEPT’,shell=True)
subprocess.call(‘iptables -A INPUT -p gre -j ACCEPT’,shell=True)
subprocess.call(‘iptables -A INPUT -p tcp -m tcp –dport 21 -j ACCEPT’,shell=True)
subprocess.call(‘iptables -A INPUT -p tcp -m state –state NEW -m tcp –dport 1723 -j ACCEPT’,shell=True)
subprocess.call(‘iptables -A FORWARD -p tcp –syn -s 192.168.1.0/24 -j TCPMSS –set-mss 1356’,shell=True)
subprocess.call(‘iptables -t nat -A POSTROUTING -s 192.168.1.0/24 -o eth0 -j MASQUERADE’,shell=True)
subprocess.call(‘service iptables save’,shell=True)
subprocess.call(‘chkconfig pptpd on’,shell=True)
subprocess.call(‘chkconfig iptables on’,shell=True)
subprocess.call(‘service iptables restart’,shell=True)
subprocess.call(‘service pptpd start’,shell=True)

二.centos下的pptpd连接客户端代码：

import subprocess
subprocess.call(‘yum -y install pptp pptp-setup’,shell=True)
subprocess.call(‘sudo modprobe nf_conntrack_pptp’,shell=True)
subprocess.call(‘pptpsetup –create VPN –server 192.168.19.187 –username user2 –password user2yourmima –encrypt’,shell=True)
subprocess.call(‘pppd call VPN’,shell=True)
subprocess.call(‘ip a |grep ppp’,shell=True)

  • 三.ubuntu下的pptpd服务器代码：

import subprocess
subprocess.call(‘sudo apt-get install pptpd -y’,shell=True)

with open(“/etc/pptpd.conf”, “r”) as f1:
d1 = f1.readlines()
with open(“/etc/pptpd.conf”, “w”) as f_w1:
for line in d1:
line = line.replace(“#localip 192.168.0.1”, “localip 192.168.1.1”)
line = line.replace(“#remoteip 192.168.0.234-238,192.168.0.245”, “remoteip 192.168.1.2-238”)
f_w1.write(line)

with open(“/etc/ppp/pptpd-options”, “r”) as f2:
d2 = f2.readlines()
with open(“/etc/ppp/pptpd-options”, “w”) as f_w2:
for line in d2:
line = line.replace(“#ms-dns 10.0.0.1”, “ms-dns 8.8.8.8”)
line = line.replace(“#ms-dns 10.0.0.2”, “ms-dns 8.8.4.4”)
f_w2.write(line)

f3=open(“/etc/ppp/chap-secrets”,’a’)
for i in range(1,101):
f3.write(“user”+str(i)+” pptpd”+” user”+str(i)+”yourmima”+” *”+”\n”)

with open(“/etc/sysctl.conf”, “r”) as f4:
d4 = f4.readlines()
with open(“/etc/sysctl.conf”, “w”) as f_w4:
for line in d4:
line = line.replace(“#net.ipv4.ip_forward=1”, “net.ipv4.ip_forward=1”)
f_w4.write(line)
subprocess.call(‘sysctl -p’,shell=True)
subprocess.call(‘/etc/init.d/procps restart’,shell=True)

subprocess.call(‘sudo apt-get install iptables’,shell=True)
subprocess.call(‘sudo iptables -A INPUT -i lo -p all -j ACCEPT’,shell=True)
subprocess.call(‘sudo iptables -A INPUT -p gre -j ACCEPT’,shell=True)
subprocess.call(‘sudo iptables -A INPUT -p tcp –dport 1723 -j ACCEPT’,shell=True)
subprocess.call(‘sudo iptables -A OUTPUT -p tcp –sport 1723 -j ACCEPT’,shell=True)
subprocess.call(‘iptables -A OUTPUT -p gre -j ACCEPT’,shell=True)
subprocess.call(‘iptables -A FORWARD -i ppp+ -j ACCEPT’,shell=True)
subprocess.call(‘iptables -A FORWARD -o ppp+ -j ACCEPT’,shell=True)
subprocess.call(‘iptables -A POSTROUTING -t nat -o ppp+ -j MASQUERADE’,shell=True)
subprocess.call(‘sudo iptables -A INPUT -p tcp -m tcp –dport 21 -j ACCEPT’,shell=True)
subprocess.call(‘sudo iptables -t nat -A POSTROUTING -s 192.168.1.0/24 -o ens33 -j MASQUERADE’,shell=True)
subprocess.call(‘sudo iptables -A INPUT -p tcp -m state –state NEW -m tcp –dport 22 -j ACCEPT’,shell=True)
subprocess.call(‘sudo iptables -A INPUT -p tcp -m state –state NEW -m tcp –dport 1723 -j ACCEPT’,shell=True)
subprocess.call(‘sudo iptables -A FORWARD -p tcp –syn -s 192.168.1.0/24 -j TCPMSS –set-mss 1356’,shell=True)
subprocess.call(‘sudo iptables -A POSTROUTING -t nat -s 192.168.1.0/24 -j SNAT –to 192.168.19.156’,shell=True)
subprocess.call(‘sudo iptables -t nat -A POSTROUTING -s 192.168.1.0/24 -o ens33 -j MASQUERADE’,shell=True)
f5=open(“/etc/network/interfaces”,’a’)
f5.write(“pre-up iptables-restore </etc/iptables-rules”)
subprocess.call(‘sudo iptables-save >/etc/iptables-rules’,shell=True)
subprocess.call(‘sudo /etc/init.d/procps restart’,shell=True)
subprocess.call(‘sudo /etc/init.d/pptpd restart’,shell=True)
